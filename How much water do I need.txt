function howMuchWater(water, load, clothes){
  // Insert your brilliant code here
  if (clothes > (load*2)) {
    return "Too much clothes";
  } else if (clothes < load) {
    return "Not enough clothes";
  } else {
    return water * Math.pow(1.1, (clothes - load));
  }
}
